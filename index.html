'); responseCache.set(message, reply); // 存入缓存 responseDiv.innerHTML = reply; } catch (error) { responseDiv.innerHTML = `错误: ${error.message}`; } input.value = ''; // 清空输入框 } // 添加防抖 const debouncedSendMessage = debounce(sendMessage, 300); // 更新按钮和 Enter 键监听 document.querySelector('.chat-button').onclick = debouncedSendMessage; document.getElementById('chatInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') debouncedSendMessage(); });
